30 def hillshade(array, azimuth, angle_altitude):
31  x, y = np.gradient(array)
32  slope = np.pi / 2. - np.arctan(np.sqrt(x * x + y * y))
33  aspect = np.arctan2(-x, y)
34  azimuthrad = azimuth * np.pi / 180.
35  altituderad = angle_altitude * np.pi / 180.
36 
37  shaded = np.sin(altituderad) * np.sin(slope) + np.cos(altituderad) * np.cos(slope) * np.cos(azimuthrad - aspect)
38  return 255 * (shaded + 1) / 2
